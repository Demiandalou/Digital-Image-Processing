def hough(img_path):
    img = cv2.imread(img_path)
    img = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
    # canny
    edges = cv2.Canny(img,50,150, apertureSize = 3)
    # hough line detection
    lines = cv2.HoughLines(edges,1,np.pi/180,150)
    for line in lines:
        for rho,theta in line:
            a = np.cos(theta) 
            b = np.sin(theta) 
            x0 = a*rho 
            y0 = b*rho 
            x1 = int(x0 + 1000*(-b)) 
            y1 = int(y0 + 1000*(a)) 
            x2 = int(x0 - 1000*(-b)) 
            y2 = int(y0 - 1000*(a))
#             print(x1, y1, x2, y2)
            cv2.line(img,(x1,y1),(x2,y2),255,2)
#     cv2.imwrite('houghlines3.jpg',img)



    plt.figure()
    plt.subplot(1, 2, 1); plt.imshow(edges, cmap='gray'); plt.title('canny'); plt.show()
    plt.subplot(1, 2, 2); plt.imshow(img, cmap='gray'); plt.title('line hough transform');plt.show()
